#include <stdio.h>
#include <string.h>

struct academicdetails {
    char name[50];  
    int regno;
    char department[50];
    int marks[5];
};

struct personaldetails {
    int age;
    char address[100];
    char phoneno[15];
};

struct student {
    struct academicdetails academic;
    struct personaldetails personal;
};

void inputstudentdetails(struct student *student) {
    printf("Enter register number: ");
    scanf("%d", &student->academic.regno);
    printf("Enter name: ");
    scanf(" %[^\n]s", student->academic.name);  
    printf("Enter Department: ");
    scanf(" %[^\n]s", student->academic.department);  
    printf("Enter 5 subject marks: ");
    for (int i = 0; i < 5; i++) {
        scanf("%d", &student->academic.marks[i]);
    }
    printf("Enter age: ");
    scanf("%d", &student->personal.age);
    printf("Enter address: ");
    scanf(" %[^\n]s", student->personal.address);  
    printf("Enter phone number: ");
    scanf("%s", student->personal.phoneno);
}

void displaystudentdetails(struct student *student) {
    printf("\n\nStudent Details: \n");
    printf("Reg number: %d\n", student->academic.regno);
    printf("Name: %s\n", student->academic.name);
    printf("Department: %s\n", student->academic.department);
    printf("Subject marks: ");
    for (int i = 0; i < 5; i++) {
        printf("%d ", student->academic.marks[i]);
    }
    printf("\n");
    printf("\n\nPersonal Details: \n");
    printf("Age: %d\n", student->personal.age);
    printf("Address: %s\n", student->personal.address);
    printf("Phone number: %s\n", student->personal.phoneno);
}

int main() {
    int n;
    printf("Enter the number of students: ");
    scanf("%d", &n);
    struct student students[n];
    for (int i = 0; i < n; i++) {
        printf("Entering details for student %d\n", i + 1);
        inputstudentdetails(&students[i]);
    }
    for (int i = 0; i < n; i++) {
        displaystudentdetails(&students[i]);
    }
    return 0;
}

// ->: The arrow operator is used to access members of a structure through a pointer.
